HTML Best Practices

Document Structure

Following proper HTML structure ensures better maintainability, accessibility, and SEO.

Basic Document Structure


<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Page Title</title>
    <link rel="stylesheet" href="styles.css">
</head>
<body>
    <header>
        <nav>
            <!-- Navigation content -->
        </nav>
    </header>

    <main>
        <article>
            <section>
                <!-- Main content sections -->
            </section>
        </article>
        <aside>
            <!-- Sidebar content -->
        </aside>
    </main>

    <footer>
        <!-- Footer content -->
    </footer>
</body>
</html>

Code Organization

Clean and Consistent Code


<!-- DO: Use consistent indentation -->
<article>
    <h1>Title</h1>
    <section>
        <h2>Subtitle</h2>
        <p>Content here...</p>
    </section>
</article>

<!-- DON'T: Mix indentation or formatting -->
<article>
<h1>Title</h1>
  <section>
    <h2>Subtitle</h2>
      <p>Content here...</p>
  </section>
</article>

<!-- DO: Use meaningful class names -->
<div class="featured-article">
    <h2 class="article-title">Title</h2>
    <p class="article-excerpt">Excerpt...</p>
</div>

<!-- DON'T: Use generic names -->
<div class="div1">
    <h2 class="title1">Title</h2>
    <p class="text1">Excerpt...</p>
</div>

Performance Optimization

Resource Loading


<!-- DO: Optimize resource loading -->
<head>
    <!-- Critical CSS inline -->
    <style>
        /* Critical styles here */
    </style>

    <!-- Defer non-critical CSS -->
    <link 
        rel="stylesheet" 
        href="styles.css"
        media="print" 
        onload="this.media='all'"
    >

    <!-- Defer non-critical JavaScript -->
    <script src="app.js" defer></script>

    <!-- Preload important resources -->
    <link 
        rel="preload" 
        href="critical-font.woff2" 
        as="font" 
        type="font/woff2" 
        crossorigin
    >
</head>

<!-- DO: Optimize images -->
<picture>
    <source 
        srcset="image-small.webp" 
        media="(max-width: 768px)" 
        type="image/webp"
    >
    <source 
        srcset="image-large.webp" 
        type="image/webp"
    >
    <img 
        src="image-fallback.jpg" 
        alt="Description"
        loading="lazy"
    >
</picture>

Security Best Practices

Content Security


<!-- DO: Use secure attributes -->
<form action="/submit" method="POST">
    <!-- Prevent autocomplete on sensitive fields -->
    <input 
        type="password" 
        autocomplete="new-password"
        id="password"
    >

    <!-- Use CSRF tokens -->
    <input 
        type="hidden" 
        name="csrf_token" 
        value="token_here"
    >
</form>

<!-- DO: Secure external resources -->
<img 
    src="https://example.com/image.jpg" 
    crossorigin="anonymous"
    referrerpolicy="no-referrer"
>

<!-- DO: Use secure links -->
<a 
    href="https://example.com" 
    rel="noopener noreferrer" 
    target="_blank"
>
    External Link
</a>
