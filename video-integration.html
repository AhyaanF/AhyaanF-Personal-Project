Web Video Integration

HTML5 Video

Learn how to embed videos using native HTML5 capabilities.

Basic Video Implementation


1. Basic Video Element
   <video width="720" height="480" controls>
     <source src="video.mp4" type="video/mp4">
     <source src="video.webm" type="video/webm">
     Your browser doesn't support HTML5 video.
   </video>

2. Video Attributes
   - controls (playback controls)
   - autoplay (automatic playback)
   - loop (continuous playback)
   - muted (start muted)
   - preload (loading behavior)

3. Responsive Design
   <div class="video-container">
     <video width="100%" height="auto">
       <source src="video.mp4" type="video/mp4">
     </video>
   </div>

   .video-container {
     position: relative;
     padding-bottom: 56.25%; /* 16:9 */
     height: 0;
     overflow: hidden;
   }
   .video-container video {
     position: absolute;
     top: 0;
     left: 0;
     width: 100%;
     height: 100%;
   }

Video Platforms

Integrate videos from popular streaming platforms.

Platform Integration


1. YouTube Embed
   <iframe 
     width="560" 
     height="315" 
     src="https://www.youtube.com/embed/VIDEO_ID"
     frameborder="0" 
     allowfullscreen>
   </iframe>

2. Vimeo Embed
   <iframe 
     src="https://player.vimeo.com/video/VIDEO_ID" 
     width="640" 
     height="360" 
     frameborder="0" 
     allowfullscreen>
   </iframe>

3. Custom Player
   <div id="player"></div>
   <script src="https://player.platform.com/api.js"></script>
   <script>
     const player = new Player('player', {
       video: 'VIDEO_ID',
       width: 640,
       height: 360
     });
   </script>

Advanced Features

Implement advanced video features and optimizations.

Advanced Implementation


1. Lazy Loading
   <video 
     loading="lazy"
     poster="thumbnail.jpg"
     data-src="video.mp4">
   </video>

   <script>
   const observer = new IntersectionObserver((entries) => {
     entries.forEach(entry => {
       if (entry.isIntersecting) {
         const video = entry.target;
         video.src = video.dataset.src;
         observer.unobserve(video);
       }
     });
   });
   </script>

2. Custom Controls
   <div class="video-player">
     <video id="myVideo">
       <source src="video.mp4" type="video/mp4">
     </video>
     <div class="controls">
       <button onclick="playPause()">Play/Pause</button>
       <input type="range" id="progress" value="0">
       <button onclick="toggleMute()">Mute</button>
     </div>
   </div>

3. Adaptive Streaming
   <video>
     <source src="video-high.mp4" type="video/mp4" media="(min-width: 800px)">
     <source src="video-medium.mp4" type="video/mp4" media="(min-width: 400px)">
     <source src="video-low.mp4" type="video/mp4">
   </video>

Practice Exercise

Create a responsive video player with:

  • Custom controls
  • Lazy loading
  • Multiple video sources
  • Fallback content

<!-- HTML Structure -->
<div class="video-wrapper">
  <div class="video-container">
    <video id="customPlayer" poster="thumbnail.jpg" preload="none">
      <source data-src="video-hd.mp4" type="video/mp4" media="(min-width: 1080px)">
      <source data-src="video-sd.mp4" type="video/mp4">
      <track kind="captions" src="captions.vtt" srclang="en" label="English">
      <p>Your browser doesn't support HTML5 video.</p>
    </video>
    
    <div class="custom-controls">
      <button class="play-pause">Play</button>
      <div class="progress-bar">
        <div class="progress"></div>
      </div>
      <button class="mute">Mute</button>
      <select class="quality">
        <option value="hd">HD</option>
        <option value="sd">SD</option>
      </select>
    </div>
  </div>
</div>

<!-- CSS Styles -->
<style>
.video-wrapper {
  max-width: 1200px;
  margin: 0 auto;
}

.video-container {
  position: relative;
  padding-bottom: 56.25%;
  height: 0;
  overflow: hidden;
}

.video-container video {
  position: absolute;
  top: 0;
  left: 0;
  width: 100%;
  height: 100%;
}

.custom-controls {
  position: absolute;
  bottom: 0;
  width: 100%;
  background: rgba(0,0,0,0.5);
  padding: 10px;
}
</style>

<!-- JavaScript -->
<script>
document.addEventListener('DOMContentLoaded', function() {
  const video = document.getElementById('customPlayer');
  const playPauseBtn = document.querySelector('.play-pause');
  const progress = document.querySelector('.progress');
  
  // Lazy loading
  const observer = new IntersectionObserver((entries) => {
    entries.forEach(entry => {
      if (entry.isIntersecting) {
        const sources = video.querySelectorAll('source');
        sources.forEach(source => {
          source.src = source.dataset.src;
        });
        video.load();
        observer.unobserve(video);
      }
    });
  });
  
  observer.observe(video);
  
  // Custom controls
  playPauseBtn.addEventListener('click', () => {
    if (video.paused) {
      video.play();
      playPauseBtn.textContent = 'Pause';
    } else {
      video.pause();
      playPauseBtn.textContent = 'Play';
    }
  });
  
  // Update progress bar
  video.addEventListener('timeupdate', () => {
    const percentage = (video.currentTime / video.duration) * 100;
    progress.style.width = percentage + '%';
  });
});
</script>
