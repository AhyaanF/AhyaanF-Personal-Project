HTML5 Features

Native Form Features

HTML5 introduced many new input types and form validation features.

New Input Types


<form>
    <!-- Date and Time inputs -->
    <input type="date" name="date">
    <input type="time" name="time">
    <input type="datetime-local" name="datetime">
    <input type="month" name="month">
    <input type="week" name="week">

    <!-- Number and Range inputs -->
    <input type="number" min="0" max="100" step="5">
    <input type="range" min="0" max="100" value="50">

    <!-- Color picker -->
    <input type="color" name="theme">

    <!-- Email, URL, and Search -->
    <input type="email" name="email">
    <input type="url" name="website">
    <input type="search" name="query">

    <!-- Built-in validation -->
    <input 
        type="email" 
        required 
        pattern="[a-z0-9._%+-]+@[a-z0-9.-]+\.[a-z]{2,}$"
    >
</form>

Storage and APIs

Web Storage


<!-- Example using localStorage -->
<script>
// Store data
localStorage.setItem('username', 'John');
localStorage.setItem('theme', 'dark');

// Retrieve data
const username = localStorage.getItem('username');
const theme = localStorage.getItem('theme');

// Remove data
localStorage.removeItem('username');

// Clear all data
localStorage.clear();
</script>

<!-- Example using sessionStorage -->
<script>
// Stores data for the session
sessionStorage.setItem('tempData', 'value');
</script>

Geolocation API


<script>
if (navigator.geolocation) {
    navigator.geolocation.getCurrentPosition(
        function(position) {
            const latitude = position.coords.latitude;
            const longitude = position.coords.longitude;
            console.log(`Lat: ${latitude}, Long: ${longitude}`);
        },
        function(error) {
            console.error('Error getting location:', error);
        }
    );
}
</script>

Practice Exercise

Create a modern HTML5 page that includes:

  • A form with various HTML5 input types
  • A video player with captions
  • Local storage functionality
  • Semantic structure

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Modern HTML5 Page</title>
</head>
<body>
    <header>
        <h1>Modern HTML5 Features Demo</h1>
        <nav>
            <ul>
                <li><a href="#form">Form</a></li>
                <li><a href="#media">Media</a></li>
            </ul>
        </nav>
    </header>

    <main>
        <section id="form">
            <h2>Registration Form</h2>
            <form id="registration">
                <label for="name">Name:</label>
                <input type="text" id="name" required>

                <label for="email">Email:</label>
                <input type="email" id="email" required>

                <label for="birth">Birthday:</label>
                <input type="date" id="birth">

                <label for="favorite-color">Favorite Color:</label>
                <input type="color" id="favorite-color">

                <button type="submit">Submit</button>
            </form>
        </section>

        <section id="media">
            <h2>Media Player</h2>
            <video controls width="640" height="360">
                <source src="video.mp4" type="video/mp4">
                <track 
                    kind="captions" 
                    src="captions.vtt" 
                    srclang="en" 
                    label="English"
                >
            </video>
        </section>
    </main>

    <aside>
        <h2>Saved Data</h2>
        <div id="saved-data"></div>
    </aside>

    <footer>
        <p>© 2024 HTML5 Demo</p>
    </footer>

    <script>
        // Form handling and local storage
        document.getElementById('registration')
            .addEventListener('submit', function(e) {
                e.preventDefault();
                
                const name = document.getElementById('name').value;
                const email = document.getElementById('email').value;
                
                // Save to localStorage
                localStorage.setItem('userData', JSON.stringify({
                    name: name,
                    email: email
                }));
                
                // Update display
                displaySavedData();
            });

        function displaySavedData() {
            const userData = JSON.parse(
                localStorage.getItem('userData')
            );
            if (userData) {
                document.getElementById('saved-data').innerHTML = `
                    <p>Name: ${userData.name}</p>
                    <p>Email: ${userData.email}</p>
                `;
            }
        }

        // Display saved data on load
        displaySavedData();
    </script>
</body>
</html>
